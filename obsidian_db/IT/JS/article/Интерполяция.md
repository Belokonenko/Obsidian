```javascript
// Использование переменной в строке
const name = "Alice";
console.log(`Привет, ${name}!`);

// Использование выражения в строке
const a = 10;
const b = 5;
console.log(`Сумма ${a} и ${b} равна ${a + b}.`);

// Вложенная интерполяция
const x = 3;
const y = 4;
console.log(`Расстояние от точки (${x}, ${y}) до начала координат равно ${Math.sqrt(x ** 2 + y ** 2)}.`);
```
